        function init() {
            // Check local storage for user
            const storedUser = localStorage.getItem('tradeIQ_user');
            if (storedUser) {
                state.user = JSON.parse(storedUser);
                updateAuthUI();

        function router(page, params = {}) {
            state.currentPage = page;
            const app = document.getElementById('app');
            window.scrollTo(0,0);

            // Update Active Link
            document.querySelectorAll('.nav-links a').forEach(el => el.classList.remove('active'));
            
            if (page === 'home') {
                app.innerHTML = renderHome();
                document.querySelector('.nav-links a:nth-child(1)').classList.add('active');
            } else if (page === 'courses') {
                app.innerHTML = renderCourses();
                document.querySelector('.nav-links a:nth-child(2)').classList.add('active');
            } else if (page === 'lesson') {
                app.innerHTML = renderLesson(params.courseId, params.lessonId);
            } else if (page === 'dashboard') {
                if(!state.user) { toggleModal('authModal'); router('home'); return; }
                app.innerHTML = renderDashboard();
                document.getElementById('nav-dashboard').classList.add('active');
            } else if (page === 'blog') {
                app.innerHTML = renderBlog();
                document.querySelector('.nav-links a:nth-child(3)').classList.add('active');
            }
        }

        function checkAnswer(el, selectedIdx, correctIdx) {
            // Reset styles

            options.forEach(opt => opt.classList.remove('selected', 'correct', 'wrong'));

            // Select clicked
            el.classList.add('selected');

            const feedback = document.getElementById('quizFeedback');
            if(selectedIdx === correctIdx) {
                el.classList.add('correct');
                feedback.textContent = "Correct! Well done.";
                feedback.style.color = "var(--accent)";
            } else {
                el.classList.add('wrong');
                options[correctIdx].classList.add('correct'); // Show correct one
                feedback.textContent = "Incorrect. Review the lesson content.";
                feedback.style.color = "var(--danger)";
            }
        }
